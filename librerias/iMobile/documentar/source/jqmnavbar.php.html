 * @author Jose Alexis Correa Valencia
 * @copyright Copyright (c) 2016, Jose Alexis Correa Valencia
 * @license http://www.gnu.org/licenses/gpl.html     * @param bool $returnAdded

    function addButton($text, $href='', $theme='', $icon='', $returnAdded=false) {
        $bt = $this->add(new jqmButton(), true);
        $bt->text($text)->href($href)->theme($theme)->icon($icon);
        if ($returnAdded) return $bt;
        return $this;
    }
    /**
     * @access private
     * @param string $value
     * @return string|jqmContent
     */
    function role() {
        if (func_num_args() == 0) return $this->_role->value();
        $this->_role->value(func_get_arg(0));
        return $this;
    }
}
